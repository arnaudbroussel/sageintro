USE `sql11164259`;

DROP TABLE IF EXISTS `customer`;
CREATE TABLE `customer` (
  `CUSTOMER_ID` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `CUSTOMER_CODE` varchar(100) NOT NULL,
  `CUSTOMER_NAME` varchar(200) NOT NULL,
  PRIMARY KEY (`CUSTOMER_ID`) USING BTREE,
  UNIQUE KEY `UNI_CUSTOMER_ID` (`CUSTOMER_CODE`) USING BTREE
) ENGINE=InnoDB AUTO_INCREMENT=34 DEFAULT CHARSET=utf8;


DROP TABLE IF EXISTS `invoice`;
CREATE TABLE `invoice` (
  `INVOICE_ID` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `INVOICE_NUMBER` int(10) unsigned NOT NULL,  
  `INVOICE_DESCRIPTION` varchar(500) NOT NULL,
  `INVOICE_DATE` date NOT NULL,
  `INVOICE_AMOUNT` float(10,2) DEFAULT NULL,
  `INVOICE_STATUS` varchar(1) NOT NULL,
  `CUSTOMER_ID` int(10) unsigned NOT NULL,
  PRIMARY KEY (`INVOICE_ID`) USING BTREE,
  UNIQUE KEY `UNI_INVOICE_NUMBER` (`INVOICE_NUMBER`),
  KEY `FK_CUSTOMER_TRANSACTION_CUSTOMER_ID` (`CUSTOMER_ID`),
  CONSTRAINT `FK_CUSTOMER_TRANSACTION_CUSTOMER_ID` FOREIGN KEY (`CUSTOMER_ID`)
  REFERENCES `customer` (`CUSTOMER_ID`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=36 DEFAULT CHARSET=utf8;
